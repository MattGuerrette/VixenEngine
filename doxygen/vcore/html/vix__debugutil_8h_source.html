88  //NOTE:
89  // This is declared as an "assumed" virtual definition of the
90  // debug print function, that way we can make different version
91  // of the same print function if we so choose
92  //
93  // Reason:
94  // It is impossible to pass va_list around so
95  // we must create a new function with new args if
96  // we want to have different params
97  //
98  inline int VDebugPrintF(const UChar* format, va_list argList)
99  {
100  static UChar s_buffer[VIX_BUFSIZE];
101  int written = -1;
102 
103 #ifdef VIX_SYS_WINDOWS
104  #ifdef UNICODE
105  written = _vsnwprintf_s(s_buffer, VIX_BUFSIZE, format, argList);
106  #else
107  written = vsnprintf_s(s_buffer, VIX_BUFSIZE, format, argList);
108  #endif
109 #else
110  written = vsnprintf(s_buffer, sizeof(s_buffer), format, argList);
111 #endif
112 
113  //Call Win32 API with formatted string
114 #ifdef VIX_SYS_WINDOWS
115  OutputDebugString(s_buffer);
116  ConsoleWriteErr(UString(s_buffer));
117 #else
118  std::cerr << s_buffer << std::endl;
119 #endif
120  return written;
121  }

123  //NOTE:
124  // This version of the VDebugPrintF takes an ellipses
125  // as the argList param, allowing us to specify as many
126  // parameters as we want in the buffer
127  //
128  inline int DebugPrintF(const UChar* format, ...)
129  {
130  va_list argList;
131  va_start(argList, format);
132 
133  int written = VDebugPrintF(format, argList);
134 
135  va_end(argList);
136  return written;
137  }
